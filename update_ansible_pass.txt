#!/bin/bash



# Authenticate with AppRole
VAULT_TOKEN=$(curl -s --request POST \
  --header "X-Vault-Namespace: $VAULT_NAMESPACE" \
  --header "Content-Type: application/json" \
  --data "{\"role_id\":\"${APPROLE_ID}\",\"secret_id\":\"${SECRET_ID}\"}" \
  ${VAULT_ADDR}/v1/auth/approle/login | jq -r '.auth.client_token')

# Check if the token was successfully obtained
if [[ -z "$VAULT_TOKEN" || "$VAULT_TOKEN" == "null" ]]; then
  echo "Error: Failed to authenticate with Vault."
  exit 1
fi

echo "Successfully authenticated with Vault. Token: $VAULT_TOKEN"

# Set the path to the secret and the new value
SECRET_PATH="ansiblekv2/data/ansiblepass"
NEW_PASSWORD="NewSecurePassword"  ## Here you need to pass the new password value

# Read the existing secret to preserve other keys
EXISTING_SECRET=$(curl -s --request GET \
  --header "X-Vault-Namespace: $VAULT_NAMESPACE" \
  --header "X-Vault-Token: $VAULT_TOKEN" \
  $VAULT_ADDR/v1/$SECRET_PATH | jq -r '.data.data')

echo "The value of 'ansible_pass' is: $EXISTING_SECRET"

# Update the key while preserving others
UPDATED_SECRET=$(echo "$EXISTING_SECRET" | jq --arg newpass "$NEW_PASSWORD" '.ansible_pass = $newpass')

# Write the updated secret back to Vault
UPDATE_RESPONSE=$(curl -s --request POST \
  --header "X-Vault-Namespace: $VAULT_NAMESPACE" \
  --header "X-Vault-Token: $VAULT_TOKEN" \
  --header "Content-Type: application/json" \
  --data "{\"data\":$UPDATED_SECRET}" \
  $VAULT_ADDR/v1/ansiblekv2/data/ansiblepass)

# Check if the update was successful
if [[ $(echo "$UPDATE_RESPONSE" | jq -r '.errors') != "null" ]]; then
  echo "Error: Failed to update the secret."
  echo "Response: $UPDATE_RESPONSE"
  exit 1
fi

echo "Successfully updated the 'ansible_pass' key to: $NEW_PASSWORD"
